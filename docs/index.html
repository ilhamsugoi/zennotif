// No API tokens needed — uses session cookies
async function fetchTickets(subdomain, viewId) {
  const response = await fetch(
    `https://${subdomain}.zendesk.com/api/v2/views/${viewId}/tickets.json`,
    { credentials: 'include', cache: 'no-cache' }
  );
  
  if (!response.ok) {
    if (response.status === 401) {
      // Session expired — prompt re-login
      await chrome.storage.local.set({ 
        connectionStatus: 'expired' 
      });
    }
    return null;
  }
  
  const data = await response.json();
  return data.tickets;
}
// Snapshot-based change detection
const previousMap = storedSnapshot || {};
const newTickets = [];

for (const ticket of currentTickets) {
  const prevStatus = previousMap[ticket.id];
  const currentStatus = ticket.status;
  
  if (prevStatus === undefined) {
    // New ticket in view
    newTickets.push({
      id: ticket.id,
      subject: ticket.subject,
      reason: 'New ticket'
    });
  } else if (prevStatus !== currentStatus) {
    // Status changed
    newTickets.push({
      id: ticket.id,
      subject: ticket.subject,
      reason: `Changed to ${currentStatus}`
    });
  }
}

// Update snapshot for next comparison
await chrome.storage.local.set({
  [`snapshot_${viewId}`]: currentMap
});
// Manifest V3 requires offscreen document for audio
async function ensureOffscreen() {
  const contexts = await chrome.runtime.getContexts({
    contextTypes: ['OFFSCREEN_DOCUMENT']
  });
  
  if (contexts.length === 0) {
    await chrome.offscreen.createDocument({
      url: 'offscreen.html',
      reasons: ['AUDIO_PLAYBACK'],
      justification: 'Play ticket notifications'
    });
  }
}

// Send play command to offscreen
chrome.runtime.sendMessage({
  target: 'offscreen',
  type: 'play-sound',
  url: chrome.runtime.getURL('notifikasi.mp3'),
  volume: 0.8,
  tone: 'default'
});
